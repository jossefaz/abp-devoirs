PROCEDURE ProduitMatrice (A:TYPEMAT,B:TYPEMAT, VARIABLE C:TYPEMAT, DIM:ENTIER);
VARIABLE
    I,J, K : ENTIER;
DEBUT {ProduitMatrice}
    I <-0;
    J <-0;
    K <-0;

    POUR I VARIANT DE 1 A DIM FAIRE
        POUR J VARIANT DE 1 A DIM FAIRE
            C [I,J] <- 0;
            POUR I VARIANT DE K A DIM FAIRE
                C [I,J] <- A[I,J] * B[I,J];
            FAIT;
        FAIT; 
    FAIT;

FIN; {ProduitMatrice}
