PROGRAMME Exercice3;
VARIABLE 
    A : ENTIER;
    B : ENTIER;
    OPERATION : CHAINE DE CARACTERE;
DEBUT
    ECRIRE ('Entrez la valeur de A : ');
    LIRE(A);

    TANT QUE (A <= 0) FAIRE 
        ECRIRE ('Erreur de saisie : "A" DOIT etre un entier positif, veuillez ressaisir');
        LIRE(A);
    FAIT;

    ECRIRE ('Entrez la valeur de B de sorte qu elle soit inferieur a A, precedement saisi : ');
    LIRE(B);

    TANT QUE (B <= 0) FAIRE 
        ECRIRE ('Erreur de saisie : "B" DOIT etre un entier positif, veuillez ressaisir');
        LIRE(B);
    FAIT;

    TANT QUE (A < B ) FAIRE 
        ECRIRE ('Erreur de saisie : "A" DOIT etre superieur a "B", veuillez ressaisir');
        ECRIRE ('Entrez la valeur de A : ');
        LIRE(A);

        TANT QUE (A <= 0) FAIRE 
            ECRIRE ('Erreur de saisie : "A" DOIT etre un entier positif, veuillez ressaisir');
            LIRE(A);
        FAIT;

        ECRIRE ('Entrez la valeur de B : ');
        LIRE(B);

        TANT QUE (B <= 0) FAIRE 
            ECRIRE ('Erreur de saisie : "B" DOIT être un entier positif, veuillez ressaisir');
            LIRE(B);
        FAIT;
    FAIT;

    ECRIRE ('Entrez l'operation souhaitée Addition/Soustraction/Multiplication' :');
    LIRE(OPERATION);

    TANT QUE (OPERATION NON "Addition" OU OPERATION NON "Soustraction" OU OPERATION NON "Multiplication") FAIRE 
        ECRIRE ('Erreur de saisie ne rentrer que l'un des choix suivants : Addition ou Soustraction ou Multiplication :');
        LIRE(OPERATION);
    FAIT;
    
    SELON QUE OPERATION VAUT
        "Addition" : 
            ECRIRE('A + B = ' & A + B);
        "Soustraction":
            ECRIRE('A - B = ' & A - B);
        "Multiplication" :
            ECRIRE('A x B = ' & A * B);
FIN;
