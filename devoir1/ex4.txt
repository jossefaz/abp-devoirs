###### PROGRAMME 1 : INSTRUCTION POUR ######

PROGRAMME Exercice4_POUR;

VARIABLE
    N : ENTIER;
    SOMME : ENTIER;
DEBUT
    ECRIRE ('Entrez la valeur de N : ');
    LIRE(N);

    TANT QUE (NON ENTIER(N) OU N <= 0) FAIRE 
        ECRIRE ('Erreur de saisie : "N" DOIT etre un entier positif, veuillez ressaisir');
        LIRE(N);

    SOMME <- 0;
    POUR I VARIANT DE 1 A N FAIRE
        SOMME <- SOMME + I;
    FAIT;

    ECRIRE('La somme de tous les entiers de 0 a : ' & CHAINE DE CARACTERE(N) & 'est de : ' & CHAINE DE CARACTERE(SOMME) );
FIN;

###### PROGRAMME 2 : INSTRUCTION TANT QUE ######

PROGRAMME Exercice4_TANT QUE;

VARIABLE
    N : ENTIER;
    SOMME : ENTIER;
DEBUT
    ECRIRE ('Entrez la valeur de N : ');
    LIRE(N);

    TANT QUE (NON ENTIER(N) OU N <= 0) FAIRE 
        ECRIRE ('Erreur de saisie : "N" DOIT etre un entier positif, veuillez ressaisir');
        LIRE(N);

    SOMME <- 0;
    I <- 1;
    TANT QUE (I <= N) FAIRE 
        SOMME <- SOMME + I;
        I <- I +1;
    FAIT;

    ECRIRE('La somme de tous les entiers de 0 a : ' & CHAINE DE CARACTERE(N) & 'est de : ' & CHAINE DE CARACTERE(SOMME) );
FIN;

###### PROGRAMME 3 : INSTRUCTION REPETER ######

PROGRAMME Exercice4_REPETER;

VARIABLE
    N : ENTIER;
    SOMME : ENTIER;
DEBUT
    ECRIRE ('Entrez la valeur de N : ');
    LIRE(N);

    TANT QUE (NON ENTIER(N) OU N <= 0) FAIRE 
        ECRIRE ('Erreur de saisie : "N" DOIT etre un entier positif, veuillez ressaisir');
        LIRE(N);

    SOMME <- 0;
    I <- 1;
    REPETER 
        SOMME <- SOMME + I;
        I <- I +1;
    JUSQU'A (I <= N);
FIN;
